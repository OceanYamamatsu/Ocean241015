# aria2c で高速ダウンロード関数
def download_model(url, output_path):
    if "huggingface.co" in url:
        !aria2c --header="Authorization: Bearer $HUGGINGFACE_TOKEN" -o {output_path} {url}
    else:
        !aria2c -o {output_path} {url}
